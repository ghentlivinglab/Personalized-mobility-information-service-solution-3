angular.module('vopRoute').controller('routeListController', function ($scope, $http, $routeParams, $location, EventType, Route) {
 
    var entries = TransportationType.query(function() {
        var transportations = {};
        $http.get('app/json/translations.json').success(function(json) {
            angular.forEach(entries, function(entry){
                if(entry.type == "DEFAULT"){
                    var i = 0;
                    angular.forEach(entry.relevant_for_transportation_types, function(transport){
                        var tmp = {};
                        tmp["transport"] = transport;
                        tmp["transport_translated"] = json.dutch.transportation_options[transport];
                        transportations[i] = tmp;
                        i++;
                    });
                    $scope.transportations_types = transportations;
                }
            });
        });
    });
 
	$scope.routeList = Route.query({userId: $routeParams.userId, travelId: $routeParams.travelId});
 
    /**
     @description The function will got to the selected route.
     @function goto
     @memberof angular_module.vopApp.routeListController
     @param routeid: the id of the route to which we want to go.
     */
    $scope.goto = function(routeid){
        var path = $location.path();
        $location.path(path + routeid);
    };
 
    /**
     @description The function will got to the edit page of the selected route.
     @function editRoute
     @memberof angular_module.vopApp.routeListController
     @param routeid: the id of the route to which we want to edit.
     */
    $scope.editRoute = function(routeId){
        var path = $location.path();
        $location.path(path + routeId + "/edit");
    };
 
    /**
     @description The function will delete the selected route.
     @function deleteRoute
     @memberof angular_module.vopApp.routeListController
     @param routeid: the id of the route to which we want to delete.
     */
    $scope.deleteRoute = function(route) {
        var path = $location.path();
		route.$delete({userId: $routeParams.userId, travelId: $routeParams.routeId});
        if($scope.routeList.$isEmpty()){
            $location.path("/user/" + $routeParams.userId);
        } else {
            $location.path(path);
        }
 
    }
});
