 @memberof angular_module.myApp
 @constructor poiListController

angular.module('vopPOI').controller('poiListController', function($scope, $http, $location, $routeParams, PointOfInterest) {
    $scope.pois = PointOfInterest.query({userId: $routeParams.userId});
 
    /**
     @description The function will redirect to the details page of the poi
     @function goto
     @memberof angular_module.myApp.poiListController
     @param poiId - The id of the POI where to go to
     */
    $scope.goto = function(poiId){
        console.log("second");
        $location.path('/user/' + $routeParams.userId + '/poi/' + poiId);
    };
 
    /**
     @description The function will redirect to the edit page of the poi
     @function edit
     @memberof angular_module.myApp.poiListController
     @param poiId - The id of the POI where to go to for the edit
     */
    $scope.edit = function(poiId){
        console.log("first");
        $location.path('/user/' + $routeParams.userId + '/poi/' + poiId + '/edit');
    };
 
    /**
     @description The function will redirect to the details page of the poi
     @function delete
     @memberof angular_module.myApp.poiListController
     @param poiId - The id of the poi we wish to delete
     */
    $scope.delete = function(poi){
		poi.$delete({userId:$routeParams.userId})
        $location.path('/user/' + $routeParams.userId + '/poi/');
    };
 
    /**
     @description The function will redirect to the add new POI page
     @function addNew
     @memberof angular_module.myApp.poiListController
     */
    $scope.addNew = function(){
        $location.path('/user/' + $routeParams.userId + '/poi/add');
    }
});
