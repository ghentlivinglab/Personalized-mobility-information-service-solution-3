/**

angular.module('vopEvent').controller('eventEditController', function($scope, $http, $location, Event, $routeParams, EventType, TransportationType) {
  $scope.transportationTypes = TransportationType.query(function(){
    $scope.event = Event.get({eventId:$routeParams.eventId}, function(){
      if($scope.event.jams!=null){
        $scope.event.jams.forEach(function(j){
          j.show=true;
        });
      }
      $scope.event.relevant_for_transportation_types.forEach(function(t){
        for(i=0;i<$scope.transportationTypes.length;i++){
          if($scope.transportationTypes[i]==t){
            $scope.selectedEventTypesTemp[i] = t;
          }
        }
      });
    });
  });
 
  $scope.eventTypes = EventType.query();
  $scope.selectedEventTypesTemp=[];
 
  // $scope.selectedEventTypes = [];
  $scope.editEvent=function(){
    $scope.event.relevant_for_transportation_types=[];
    $scope.selectedEventTypesTemp.forEach(function(type){
      if(type!=false){
        $scope.event.relevant_for_transportation_types.push(type);
      }
    });
    var newJams = [];
    deletedJams.forEach(function(deleted){
      for(i=0;i<$scope.event.jams.length;i++){
        if($scope.event.jams[i]==deleted){
          delete $scope.event.jams[i];
        }
      }
    });
    $scope.event.jams.forEach(function(jj){
      if(jj!=undefined){
        newJams.push(jj);
      }
    });
    $scope.event.jams=newJams;
    $scope.event.coordinates={
      lat:addressAutocomplete.getPlace().geometry.location.lat(),
      lon:addressAutocomplete.getPlace().geometry.location.lng()
    };
    console.log($scope.event);
    // console.log($scope.selectedEventTypesTemp);
  };
  $scope.goBack=function(){
    $location.path("/event/"+$routeParams.eventId);
  };
  var deletedJams = [];
  $scope.deleteJam=function(jam){
    jam.show = false;
    deletedJams.push(jam);
  };
  var addressAutocomplete;
  $scope.initialize=function(){
    var geocoder = new google.maps.Geocoder();
    var addressInput = document.getElementById("addressAutocomplete");
    addressAutocomplete = new google.maps.places.Autocomplete(addressInput, {types: ['geocode']});
    var marker = new google.maps.Marker({
      position:{
        lat:$scope.event.coordinates.lat,
        lng:$scope.event.coordinates.lon
      }
    });
    geocoder.geocode({'location': {
      lat:$scope.event.coordinates.lat,
      lng:$scope.event.coordinates.lon
    }}, function(results, status){
      if (status === google.maps.GeocoderStatus.OK) {
        $scope.$apply(function(){
          if (results[0]) {
            $scope.event.addr = results[0].formatted_address;
          } else {
            console.log('No results found');
          }
        });
      } else {
        console.log('Geocoder failed due to: ' + status);
      }
    });
 
  }
  // Load the map each time the page is loaded
  $scope.loadScript = function() {
    if(!googleLoaded){
      var script = document.createElement('script');
      script.type = 'text/javascript';
      script.src = 'https://maps.googleapis.com/maps/api/js?libraries=places&sensor=false&key=AIzaSyA1ouRfRDPJsTHjpUd1IDzCJBCCjQxlu5M';
      document.body.appendChild(script);
      googleLoaded = true;
    }
    setTimeout(function() {
      $scope.initialize();
    }, 500);
  };
});
