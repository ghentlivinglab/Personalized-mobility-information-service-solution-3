angular.module('vopPOI').controller('poiAddController', function($scope, $http, $location, $routeParams, PointOfInterest, EventType) {
  var SPECIALMARKERICON = 'assets/images/map-pin-special.svg';
  var directionsDisplay;
  var geocoder;
  var pointAutocomplete;
  var pointMarker;
  var circle = null;
 
  $scope.poi = new PointOfInterest();
  $scope.poi.active=true;
  $scope.poi.notify_for_event_types = [];
  $scope.poi.notify={email:false, cell_number:false};
 
  $scope.eventTypes = EventType.query();
  $scope.tempHolder = [];
  $scope.selectedEventTypes = [];
 
  $scope.initialize = function() {
    geocoder = new google.maps.Geocoder();
    directionsDisplay = new google.maps.DirectionsRenderer();
    $scope.mapOptions = {
      // Center on Ghent
      center: {lat: 51.0533593, lng: 3.723252},
      zoom: 14
    };
    // Make new map with mapoptions
    $scope.map = new google.maps.Map(document.getElementById('map'), $scope.mapOptions);
    directionsDisplay.setMap($scope.map);
 
    var pointInput = document.getElementById("pointAutocomplete");
    $scope.map.controls[google.maps.ControlPosition.TOP_LEFT].push(document.getElementById("pointAutocompleteForm"));
    pointAutocomplete = new google.maps.places.Autocomplete(pointInput, {types: ['geocode']});
 
    $scope.map.controls[google.maps.ControlPosition.TOP_LEFT].push(document.getElementById("radiusForm"));
 
    $(document).ready(function(){
      // Stop the default submit on enter
      $("#pointAutocompleteForm").submit(function(){
        return false;
      });
      // Tell the input field what to do if you press enter
      $("#pointAutocomplete").keyup(function(event){
        if(event.keyCode == 13){
          // Wait half a second, so we don't get errors, because not everything is loaded correctly
          setTimeout(function(){$scope.addPoint();
          },500);
        }
      });
      // Stop the default submit on enter
      $("#radiusForm").submit(function(){
        return false;
      });
      // Tell the input field what to do if you press enter
      $("#radius").keyup(function(event){
        if(event.keyCode == 13){
          // Wait half a second, so we don't get errors, because not everything is loaded correctly
          setTimeout(function(){$scope.addNewCircle();
          },500);
        }
      });
    });
  };
 
  $scope.addNewCircle = function(){
    if(circle != null){
      circle.setMap(null);
    }
    circle = new google.maps.Circle({
      map: $scope.map,
      radius: $scope.poi.radius,
      strokeWeight: 0.3,
      strokeColor: '#000000',
      fillColor: '#009fe3',
      fillOpacity: 0.35
    });
    circle.bindTo('center', pointMarker, 'position');
  };
 
  $scope.addNewMarker = function(position, icon){
    var marker = new google.maps.Marker({
      draggable: true,
      icon: icon,
      position: position,
      map: $scope.map,
      animation: google.maps.Animation.DROP
    });
    marker.addListener('click', function() {
      if (marker.getAnimation() !== null) {
        marker.setAnimation(null);
      } else {
        marker.setAnimation(google.maps.Animation.BOUNCE);
      }
    });
    var left=0;
    var right=0;
    var n = 0;
    var start;
    // Delete marker by swiping. Hacky algorithm :-)
    marker.addListener('drag', function(){
      n=n+1;
      if(start<$scope.map.getProjection().fromLatLngToPoint(marker.getPosition()).x){
        right+=1;
      } else {
        left+=1;
      }
      if(n>50){
        if(Math.abs(left-right)<25){
          marker.setMap(null);
          if(circle != null){
            circle.setMap(null);
          }
        }
      }
    });
    marker.addListener('dragend', function(){
      left=0;
      start=0;
      right=0;
      n=0;
    });
    marker.addListener('dragstart', function(){
      start=$scope.map.getProjection().fromLatLngToPoint(marker.getPosition()).x;
    });
    return marker;
  };
 
  $scope.addPoint = function(){
    if(pointMarker != null){
      pointMarker.setMap(null);
    }
    pointMarker = $scope.addNewMarker({
      lat: pointAutocomplete.getPlace().geometry.location.lat(),
      lng: pointAutocomplete.getPlace().geometry.location.lng()
    }, SPECIALMARKERICON);
    $scope.mapOptions = {
      // Center on POI!
      center: {lat: pointAutocomplete.getPlace().geometry.location.lat(), lng: pointAutocomplete.getPlace().geometry.location.lng()},
      zoom: 14
    };
    if(circle != null){
      circle.bindTo('center', pointMarker, 'position');
      circle.setMap($scope.map);
    }
  };

  /**
  @description The function will save the changes made to the POI and redirects to the show-page of the POI
  @function save
  @memberof angular_module.myApp.poiAddController
  */
  $scope.save = function(){
    if(pointMarker==null){
      $scope.showErrorMessage("Gelieve een punt van interesse op te geven.");
      return;
    }
    geocoder.geocode({'location': pointMarker.position}, function(results, status){
      if (status === google.maps.GeocoderStatus.OK) {
        if (results[0]) {
          $scope.$apply(function(){
            $scope.poi.address = formatAddress(results[0], pointMarker.getPosition());
            console.log($scope.poi.address);
            angular.forEach($scope.tempHolder, function(entry){
              if(entry!=null){
                $scope.selectedEventTypes.push(entry);
              }
            });
            $scope.poi.notify_for_event_types = $scope.selectedEventTypes;
            console.log($scope.poi);
            $scope.poi.$save({userId: $routeParams.userId}, function(response){
              console.log(response);
              $location.path('/user/' + $routeParams.userId + '/poi/');
            });
          });
          // $scope.poi.address.country = $scope.poi.address.country.substr(0,2).toUpperCase();
        } else {
          window.alert('No results found');
        }
      } else {
        window.alert('Geocoder failed due to: ' + status);
      }
    });
 
  };
  /**
  @description The function will redirect to the list page of the POIs
  @function cancel
  @memberof angular_module.myApp.poiAddController
  */
  $scope.cancel = function(){
    $location.path('/user/' + $routeParams.userId + '/poi');
  };
});
 
var formatAddress = function(address, position){
  var result={};
  result.housenumber = "1";
  address.address_components.forEach(function(entry){
    if(entry.types[0]==="route" || entry.types[0]==="premise"){
      result.street = entry.long_name;
    } else if(entry.types[0]==="street_number"){
      result.housenumber = entry.long_name;
    } else if (entry.types[0]==="locality"){
      result.city = entry.long_name;
    } else if(entry.types[0]==="country"){
      result.country = entry.short_name;
    } else if(entry.types[0]==="postal_code"){
      result.postal_code = entry.long_name;
    }
  });
  result.coordinates = {
    lat: position.lat(),
    lon: position.lng()
  };
  // var result = {

  //     lat: position.lat(),

  // }
  return result;
}
