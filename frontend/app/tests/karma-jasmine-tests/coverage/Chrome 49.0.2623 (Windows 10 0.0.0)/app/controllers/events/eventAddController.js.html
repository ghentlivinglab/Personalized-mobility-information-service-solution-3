  $scope.addJam=function(){

      var pushJam={};

        pushJam = {
          start_node:{
            lat:markerArray[0].mark.position.lat(),
            lon:markerArray[0].mark.position.lng()

            lat:markerArray[1].mark.position.lat(),

            lon:markerArray[0].mark.position.lng()

        pushJam.delay = $scope.jamjam.delay;

      $scope.makeJamForm();
