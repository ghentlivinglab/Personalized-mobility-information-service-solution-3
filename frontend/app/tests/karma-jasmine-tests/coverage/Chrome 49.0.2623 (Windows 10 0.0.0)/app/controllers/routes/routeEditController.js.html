  var MARKERICON = 'assets/images/map-marker.svg';
  var SPECIALMARKERICON = 'assets/images/map-pin-special.svg';

  $scope.initialize = function() {
    bounds = new google.maps.LatLngBounds();
    geocoder = new google.maps.Geocoder();

    directionsDisplay = new google.maps.DirectionsRenderer();
    $scope.mapOptions = {
      // Center on Ghent
      center: {lat: 51.0533593, lng: 3.723252},
      zoom: 14
    };
    // Make new map with mapoptions
    $scope.map = new google.maps.Map(document.getElementById('map'), $scope.mapOptions);
    directionsDisplay.setMap($scope.map);
    // Add the textfields for origin and destination
    var addWaypointInput = document.getElementById('waypointAutocomplete');
    addWaypointAutocomplete = new google.maps.places.Autocomplete(addWaypointInput, {types: ['geocode']});
    // Init MarkerDropper
    var markerDropperDiv = document.createElement('div');
    var markerDropper = new MarkerDropper(markerDropperDiv, $scope, $scope.map, MARKERICON);
    $scope.map.controls[google.maps.ControlPosition.TOP_RIGHT].push(markerDropperDiv);
    // Init waypointAdder
    var waypointAdderDiv = document.createElement('div');
    var waypointAdder = new WaypointAdder(waypointAdderDiv, "editRouteDiv");
    $scope.map.controls[google.maps.ControlPosition.RIGHT_TOP].push(waypointAdderDiv);
    // Init showRouteButton
    var showRouteButtonDiv = document.createElement('div');
    var showRouteButton = new ShowRouteButton(showRouteButtonDiv, this);
    $scope.map.controls[google.maps.ControlPosition.BOTTOM_CENTER].push(showRouteButtonDiv);
    $scope.map.controls[google.maps.ControlPosition.TOP_RIGHT].push(document.getElementById("mode"));
    originMarker = new google.maps.Marker({
      icon: SPECIALMARKERICON,
      draggable: false,
      map: $scope.map,
      position:{
        lat:$scope.travel.startpoint.coordinates.lat,
        lng:$scope.travel.startpoint.coordinates.lon
      },
      animation: google.maps.Animation.DROP
    });
    destinationMarker = new google.maps.Marker({
      icon: SPECIALMARKERICON,
      draggable: false,
      map: $scope.map,
      position:{
        lat:$scope.travel.endpoint.coordinates.lat,
        lng:$scope.travel.endpoint.coordinates.lon
      },
      animation: google.maps.Animation.DROP
    });
    $scope.route.waypoints.forEach(function(w){
      bounds.extend(
      $scope.addNewMarker({
        lat:w.lat,
        lng:w.lon
      }, MARKERICON).getPosition());
    });
    bounds.extend(originMarker.getPosition());
    bounds.extend(destinationMarker.getPosition());
    $scope.map.fitBounds(bounds);
    // Add listener to map
    $scope.map.addListener('click',function(){
      // Hide the div + clear the input field when you click outside the form
      $(document).ready(function(){
        $("#addWaypoint").fadeOut();
        $("#editRouteDiv").css('transition','1s');
        $("#editRouteDiv").css('-webkit-filter','blur(0px)');
        $("#waypointAutocomplete").val('');

    $scope.selectedEventTypesTemp=[];

        for(i=0;i<$scope.eventTypes.length;i++){
          if($scope.eventTypes[i].type==t.type){
            $scope.selectedEventTypesTemp[i] = t.type;
          }
        }
      });
    });
  };
 
  $scope.renderRoute = function(){
    var waypoints = [];
    if(originMarker==null){
      $scope.showErrorMessage("Gelieve eerst een vertrekpunt in te geven.");
    } else {
      if(destinationMarker == null){
        $scope.showErrorMessage("Gelieve eerst een bestemming in te geven");
      } else {
        markers.forEach(function(entry){
          if(entry.getMap()!=null){
            if(!(entry==originMarker || entry == destinationMarker)){
              waypoints.push({
                location: entry.position,
                stopover: true
              });
            }
          }
        });
 
        var mappingDictionary = {
          "auto":"DRIVING",
          "bus":"TRANSIT",
          "tram":"TRANSIT",
          "trein":"TRANSIT",
          "fiets":"BICYCLING"
        };
        var selectedMode = mappingDictionary[document.getElementById("mode").value];
        var request = {
          origin: originMarker.getPosition(),
          destination: destinationMarker.getPosition(),
          waypoints: waypoints,
          optimizeWaypoints: true,
          travelMode: google.maps.TravelMode[selectedMode]
        };
        directionsDisplay.setOptions( { suppressMarkers: true } );
        directionsService.route(request, function(result, status){
          if (status == google.maps.DirectionsStatus.OK) {
            directionsDisplay.setDirections(result);
          } else {
            // Any kind of error
            $scope.showErrorMessage("Er ging iets fout bij het berekenen van de route. Controleer of uw internetverbinding goed werkt en probeer daarna opnieuw.");
          }
        });
      }
    }
  };

      lng: addWaypointAutocomplete.getPlace().geometry.location.lng()
    }, MARKERICON);
    // Hide the add waypoint div + clear the input field
    $(document).ready(function(){
      $("#addWaypoint").fadeOut();
      $("#editRouteDiv").css('transition','1s');
      $("#editRouteDiv").css('-webkit-filter','blur(0px)');
      $("#waypointAutocomplete").val('');
    });
  };
  var markers = [];
  $scope.addNewMarker = function(position, icon){
    var marker = new google.maps.Marker({
      draggable: true,
      icon: icon,
      position: position,
      map: $scope.map,
      animation: google.maps.Animation.DROP
    });
    marker.addListener('click', function() {
      if (marker.getAnimation() !== null) {
        marker.setAnimation(null);
      } else {
        marker.setAnimation(google.maps.Animation.BOUNCE);
      }
    });
    // Delete marker by rightclick
    marker.addListener('rightclick', function(){
      marker.setMap(null);
    });
 
    markers.push(marker);
    return marker;
  };

  /**
  @description The function will update the route with the changed values.
  @function editRoute
  @memberof angular_module.vopApp.routeEditController
  */
  $scope.editRoute = function () {
    //Empty it for good measurement.
    $scope.route.notify_for_event_types = [];
    $scope.route.waypoints=[];
    $scope.route.transportation_type = document.getElementById("mode").value;
    markers.forEach(function(m){
      $scope.route.waypoints.push({
        lat:m.position.lat(),
        lon:m.position.lng()
      });
    });
    $scope.selectedEventTypesTemp.forEach(function(type){
      $scope.route.notify_for_event_types.push({
        type: type
      });
    });
    console.log($scope.route);
    //Save changes.
    $scope.route.$edit({userId: $routeParams.userId, travelId: $routeParams.travelId, routeId: $scope.route.id});
    //Go back to the route list.
    $location.path("/user/" + $routeParams.userId + "/travel/" + $routeParams.travelId);
  };
});
