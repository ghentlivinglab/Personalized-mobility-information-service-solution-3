/**
 * @author Xavier Seyssens & Dean Parmentier
 * @module routeShowController
 * @description Controller related to editing routes.
 */
angular.module('vopRoute').controller('routeShowController', function ($scope, $http, $routeParams, $location, routeShow) {
    $scope.route = Route.get({userId: $routeParams.userId, travelId: $routeParams.travelId, routeId: $routeParams.routeId});
 
    /**
     @description The function will direct the page to the edit page of this route.
     @function editRoute
     @memberof angular_module.vopApp.routeShowController
     */
    $scope.editRoute = function(){
        var path = $location.path();
        $location.path(path + "edit");
    };
 
    /**
     @description The function will delete this route.
     @function deleteRoute
     @memberof angular_module.vopApp.routeShowController
     @param routeid: the id of the route to which we want to delete.
     */
    $scope.deleteRoute = function(route) {
        var path = $location.path();
		route.$delete({userId: $routeParams.userId, travelId: $routeParams.travelId});
        $location.path("/user/" + $routeParams.userId);
    };
});
