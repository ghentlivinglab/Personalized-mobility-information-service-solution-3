angular.module('vopEvent').directive("gmapsUserEventsShow", function () {
    return {
        restrict: "A",
        scope: false,
        link: function (scope, element, attrs, ngModel) {
            var geocoder, directionsService;
            /**
             * @constant SPECIALMARKERICON
             * @description The icon for the marker
             * @type {string}
             */
            var MARKERICON = 'assets/images/map-marker.svg';
            /**
             * @constant SPECIALMARKERICON
             * @description The icon for the marker
             * @type {string}
             */
            var SPECIALMARKERICON = 'assets/images/map-pin-special.svg';
 
            geocoder = new google.maps.Geocoder();
            directionsService = new google.maps.DirectionsService();
 
            scope.mapOptions = {
                // Center on Ghent!
                center: {lat: 51.0533593, lng: 3.723252},
                zoom: 10
            };
 
            // Make new map with mapoptions
            scope.map = new google.maps.Map(element[0].querySelector('#map'), scope.mapOptions);
 
            scope.promise.then(function () {
                var infoWindow = new google.maps.InfoWindow({
                    content: "<p>" + scope.clickedEvent.description + "</p>"
                });
                var bounds = new google.maps.LatLngBounds();
                angular.forEach(scope.userEvents, function (value) {
                    var image_url = null;
                    if (value.source.icon_url != null) {
                        image_url = value.source.icon_url;
                    } else {
                        image_url = 'assets/images/gent_slim op weg_lijn_donkergrijs.png';
                    }
                    // Check if the event is still active
                    console.log(value);
                    if (value.active) {
                        if (value.jams != null) {
                            angular.forEach(value.jams, function (jam) {
                                var m1 = new google.maps.Marker({
                                    position: {
                                        lat: jam.start_node.lat,
                                        lng: jam.start_node.lon
                                    },
                                    icon: MARKERICON,
                                    draggable: false,
                                    animation: google.maps.Animation.DROP,
                                    map: scope.map
                                });
                                var m2 = new google.maps.Marker({
                                    position: {
                                        lat: jam.end_node.lat,
                                        lng: jam.end_node.lon
                                    },
                                    icon: MARKERICON,
                                    draggable: false,
                                    animation: google.maps.Animation.DROP,
                                    map: scope.map
                                });
                                bounds.extend(m1.getPosition());
                                bounds.extend(m2.getPosition());
                                m1.addListener('click', function () {
                                    geocoder.geocode({
                                        'location': {
                                            lat: jam.start_node.lat,
                                            lng: jam.start_node.lon
                                        }
                                    }, function (results, status) {
                                        if (status === google.maps.GeocoderStatus.OK) {
                                            if (results[0]) {
                                                scope.$apply(function () {
                                                    scope.clickedEvent = value;
                                                    // console.log($scope.clickedEvent.type.type);
                                                    var content = "<div id='infoWindow' style='text-align: left'><div class='row'><div class='col-sm-4'><img src='" + image_url + "' height='50' width='50'/></div><div class='col-sm-7'><p><strong>" +
                                                        value.type.type + "</strong></p></div></div><strong>Beschrijving</strong><p>" +
                                                        value.description + "</p><strong>Adres</strong><p>" + results[0].formatted_address + "</p><strong>Publicatie datum</strong><p>" + value.publication_time + "</p><strong>Gemiddelde snelheid (km/u)</strong><p>" + jam.speed + "</p><strong>Vertraging (sec)</strong><p>" + jam.delay + "</p></div>";
                                                    infoWindow.setContent(content);
                                                    infoWindow.open(scope.map, m1);
                                                });
                                            } else {
                                                console.log('No results found');
                                            }
                                        } else {
                                            console.log('Geocoder failed due to: ' + status);
                                        }
                                    });
                                });
                                m2.addListener('click', function () {
                                    geocoder.geocode({
                                        'location': {
                                            lat: jam.end_node.lat,
                                            lng: jam.end_node.lon
                                        }
                                    }, function (results, status) {
                                        if (status === google.maps.GeocoderStatus.OK) {
                                            if (results[0]) {
                                                scope.$apply(function () {
                                                    scope.clickedEvent = value;
                                                    // console.log($scope.clickedEvent.type.type);
                                                    var content = "<div id='infoWindow' style='text-align: left'><div class='row'><div class='col-sm-4'><img src='" + image_url + "' height='50' width='50'/></div><div class='col-sm-8'><p><strong>" +
                                                        value.type.type + "</strong></p></div></div><strong>Beschrijving</strong><p>" +
                                                        value.description + "</p><strong>Adres</strong><p>" + results[0].formatted_address + "</p><strong>Publicatie datum</strong><p>" + value.publication_time + "</p></div>";
                                                    infoWindow.setContent(content);
                                                    infoWindow.open(scope.map, m2);
                                                });
                                            } else {
                                                console.log('No results found');
                                            }
                                        } else {
                                            console.log('Geocoder failed due to: ' + status);
                                        }
                                    });
                                });
                                var request = {
                                    origin: m1.getPosition(),
                                    destination: m2.getPosition(),
                                    optimizeWaypoints: true,
                                    travelMode: google.maps.TravelMode["DRIVING"]
                                };
                                var directionsDisplay = new google.maps.DirectionsRenderer({
                                    polylineOptions: {
                                        strokeColor: "red"
                                    }
                                });
                                directionsDisplay.setMap(scope.map);
                                directionsDisplay.setOptions({
                                    suppressMarkers: true,
                                    preserveViewport: true
                                });
                                directionsService.route(request, function (result, status) {
                                    if (status == google.maps.DirectionsStatus.OK) {
                                        directionsDisplay.setDirections(result);
                                    } else {
                                        // Any kind of error
                                        scope.showErrorMessage("Er ging iets fout bij het berekenen van de route. Controleer of uw internetverbinding goed werkt en probeer daarna opnieuw.");
                                    }
                                });
 
                            });
                        } else {
                            var m = new google.maps.Marker({
                                position: {
                                    lat: value.coordinates.lat,
                                    lng: value.coordinates.lon
                                },
                                icon: SPECIALMARKERICON,
                                draggable: false,
                                animation: google.maps.Animation.DROP,
                                map: scope.map
                            });
                            bounds.extend(m.getPosition());
                            m.addListener('click', function () {
                                geocoder.geocode({
                                    'location': {
                                        lat: value.coordinates.lat,
                                        lng: value.coordinates.lon
                                    }
                                }, function (results, status) {
                                    if (status === google.maps.GeocoderStatus.OK) {
                                        if (results[0]) {
                                            scope.$apply(function () {
                                                scope.clickedEvent = value;
                                                // console.log($scope.clickedEvent.type.type);
                                                var content = "<div id='infoWindow' style='text-align: left'><div class='row'><div class='col-sm-4'><img src='" + image_url + "' height='50' width='50'/></div><div class='col-sm-8'><p><strong>" +
                                                    value.type.type + "</strong></p></div></div><strong>Beschrijving</strong><p>" +
                                                    value.description + "</p><strong>Adres</strong><p>" + results[0].formatted_address + "</p><strong>Publicatie datum</strong><p>" + value.publication_time + "</p></div>";
                                                infoWindow.setContent(content);
                                                infoWindow.open(scope.map, m);
                                            });
                                        } else {
                                            console.log('No results found');
                                        }
                                    } else {
                                        console.log('Geocoder failed due to: ' + status);
                                    }
                                });
 
                            });
                        }
                        scope.map.fitBounds(bounds);
                    }
                })
            });
        }
    }
});
